## **Database Schema:**

```sql
-- Share tracking
CREATE TABLE social_shares (
  id UUID PRIMARY KEY,
  user_id UUID REFERENCES auth.users(id),
  content_type TEXT, -- visa, tour, blog, achievement
  content_id UUID,
  platform TEXT, -- facebook, twitter, whatsapp, etc
  referral_code TEXT,
  shared_at TIMESTAMP DEFAULT NOW()
);

-- Referral tracking
CREATE TABLE referrals (
  id UUID PRIMARY KEY,
  referrer_id UUID REFERENCES auth.users(id),
  referred_user_id UUID REFERENCES auth.users(id),
  referral_code TEXT,
  status TEXT, -- pending, completed, rewarded
  reward_amount INTEGER,
  created_at TIMESTAMP DEFAULT NOW(),
  completed_at TIMESTAMP
);

-- Create indexes
CREATE INDEX idx_shares_user ON social_shares(user_id);
CREATE INDEX idx_shares_content ON social_shares(content_type, content_id);
CREATE INDEX idx_referrals_referrer ON referrals(referrer_id);
CREATE INDEX idx_referrals_code ON referrals(referral_code);
```
