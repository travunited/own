### 12.2 Database Queries

**Key Queries for Admin Dashboard:**

```sql
-- Dashboard Stats
SELECT 
  COUNT(*) FILTER (WHERE status = 'UNDER_REVIEW') as pending_review,
  COUNT(*) FILTER (WHERE status = 'IN_PROGRESS') as in_progress,
  COUNT(*) FILTER (WHERE status = 'APPROVED' AND DATE(updated_at) = CURRENT_DATE) as approved_today,
  COUNT(*) FILTER (WHERE created_at >= CURRENT_DATE - INTERVAL '24 hours') as applications_today
FROM visa_applications
WHERE created_at >= CURRENT_DATE - INTERVAL '30 days';

-- Revenue Today
SELECT 
  SUM(grand_total) as total_revenue,
  COUNT(*) as order_count,
  AVG(grand_total) as avg_order_value
FROM orders
WHERE status = 'SUCCESS'
  AND DATE(created_at) = CURRENT_DATE;

-- Application Queue with Filters
SELECT 
  va.id,
  va.application_number,
  va.status,
  va.created_at,
  va.grand_total,
  u.full_name as customer_name,
  vc.name as country_name,
  vt.name as visa_type_name,
  COUNT(vap.id) as traveller_count,
  s.full_name as assigned_to_name
FROM visa_applications va
JOIN users u ON va.user_id = u.id
JOIN visa_types vt ON va.visa_type_id = vt.id
JOIN visa_countries vc ON vt.country_id = vc.id
LEFT JOIN visa_applicants vap ON va.id = vap.visa_application_id
LEFT JOIN staff_users su ON va.assigned_to = su.id
LEFT JOIN users s ON su.user_id = s.id
WHERE va.status = $1 -- Filter by status
  AND ($2::uuid IS NULL OR va.assigned_to = $2) -- Filter by assigned
GROUP BY va.id, u.full_name, vc.name, vt.name, s.full_name
ORDER BY va.created_at DESC
LIMIT $3 OFFSET $4;

-- Document Review Status
SELECT 
  vd.id,
  vd.document_type,
  vd.file_url,
  vd.status,
  vd.uploaded_at,
  vap.full_name as applicant_name
FROM visa_documents vd
JOIN visa_applicants vap ON vd.visa_applicant_id = vap.id
WHERE vap.visa_application_id = $1
ORDER BY vd.uploaded_at DESC;

-- Team Performance
SELECT 
  u.full_name,
  su.role,
  COUNT(va.id) FILTER (WHERE va.status = 'APPROVED') as approved_count,
  COUNT(va.id) as total_assigned,
  AVG(EXTRACT(EPOCH FROM (va.updated_at - va.created_at))/86400) as avg_processing_days
FROM staff_users su
JOIN users u ON su.user_id = u.id
LEFT JOIN visa_applications va ON va.assigned_to = su.id
WHERE va.created_at >= CURRENT_DATE - INTERVAL '30 days'
GROUP BY u.full_name, su.role;
```

// Middleware for admin authentication
export async function adminAuthMiddleware(req: Request) {
  const session = await getSession(req);
  
  if (!session) {
    return redirect('/admin/login');
  }
  
  const user = await getUserWithRole(session.userId);
  
  if (!user || !isAdminRole(user.role)) {
    return unauthorized();
  }
  
  return user;
}
